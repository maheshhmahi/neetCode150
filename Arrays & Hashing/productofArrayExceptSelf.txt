class Solution {
    public int[] productExceptSelf(int[] nums) {
        int res[] = new int[nums.length];

        int lsum = nums[0];
        res[0] = 1;
        res[nums.length-1] = 1;
        for(int i=1; i<nums.length; i++) {
            res[i] = lsum;
            lsum *= nums[i];
        }
        int rsum = nums[nums.length-1];
        for(int i=nums.length-2; i>=0 ; i--) {
            res[i] *= rsum;
            rsum *= nums[i];

        }
        return res;
    }
}
